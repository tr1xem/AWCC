TARGET := awcc
PREFIX ?= $(HOME)/.local
SOURCE := main.c include/effects.c include/fans.c include/lights.c src/gui.c
INCLUDE_DIRS := ./include ./src

# Compiler and linker flags
CFLAGS := -Wall -Wextra $(foreach dir,$(INCLUDE_DIRS),-I$(dir)) $(shell pkg-config --cflags gtk4)
LDFLAGS := -lusb-1.0 $(shell pkg-config --libs gtk4)

all: $(TARGET)

$(TARGET): $(SOURCE)
	gcc $(CFLAGS) -o $@ $(SOURCE) $(LDFLAGS)

run: $(TARGET)
	./$(TARGET)

install: $(TARGET)
	mkdir --parents "$(PREFIX)/bin"
	cp "$(TARGET)" "$(PREFIX)/bin"

uninstall:
	rm --force "$(PREFIX)/bin/$(TARGET)"

clean:
	rm --force "$(TARGET)" $(wildcard *.o) $(wildcard src/*.o) $(wildcard include/*.o)

// --- Fan Control Callbacks ---
static void execute_fan_command(const char *fan_cmd_arg) {
    char command_string[256];
    // It's important that AWCC_COMMAND (awcc) is in a directory that pkexec can access,
    // or provide full path if known. User might need to configure Polkit.
    snprintf(command_string, sizeof(command_string), "pkexec %s %s", AWCC_COMMAND, fan_cmd_arg);
    g_print("Executing fan command: %s\n", command_string);

    GError *error = NULL;
    g_spawn_command_line_async(command_string, &error);
    if (error) {
        g_warning("Failed to execute command '%s': %s", command_string, error->message);
        // Optionally show a dialog to the user
        GtkWidget *dialog = gtk_message_dialog_new(NULL, GTK_DIALOG_MODAL, GTK_MESSAGE_ERROR, GTK_BUTTONS_CLOSE,
                                                   "Failed to execute fan command: %s\nEnsure '%s' is in system PATH and pkexec is configured.",
                                                   error->message, AWCC_COMMAND);
        g_signal_connect(dialog, "response", G_CALLBACK(gtk_window_destroy), NULL);
        gtk_widget_show(dialog);
        g_error_free(error);
    }
}

sudo apt-get update

# FAQ and TIPS

Q: How do I create a keybind for Light Toggle?

Ans: Install `awcc`. Then create a script, for example, `~/.local/bin/toggle_brightness.sh`:
```bash
#!/bin/bash
# ~/.local/bin/toggle_brightness.sh
STATE_FILE="/tmp/awcc_brightness_state.txt" # Use /tmp or a more robust location
VALUES=(0 50 100) # Brightness percentages
CURRENT_INDEX=$(cat "$STATE_FILE" 2>/dev/null || echo -1)
NEXT_INDEX=$(( (CURRENT_INDEX + 1) % ${#VALUES[@]} ))
echo $NEXT_INDEX > "$STATE_FILE"
ARG=${VALUES[$NEXT_INDEX]}
echo "Executing awcc brightness $ARG"
awcc brightness "$ARG"
```
Make it executable: `chmod +x ~/.local/bin/toggle_brightness.sh`.
Then bind this script to a key (e.g., F5) using your desktop environment's keyboard shortcut settings.

Q: I am an AMD user, and the fan commands don't work.

Ans: The tool attempts to auto-detect Intel (`AMWW`) vs AMD (`AMW3`) ACPI prefixes. If auto-detection fails or is incorrect for your model:
1.  Edit `include/fans.c`.
2.  Find the line `const char *acpi_prefix = "AMWW"; // Default to Intel`.
3.  If you have an AMD CPU and it's not working, it might be that `detectCpuVendor()` isn't correctly identifying it or your model uses a different prefix. The auto-detection changes `acpi_prefix` to `"AMW3"` if "AuthenticAMD" is found in `/proc/cpuinfo`.
4.  You can manually force it by changing the default or modifying `detectCpuVendor()`.
5.  Rebuild the project: `make clean && make`.

Q: Fan controls in the GUI don't work / ask for a password repeatedly.
Ans: Fan controls require root privileges. The GUI uses `pkexec` to run `awcc` with these privileges.
    - Ensure `awcc` is installed in a location where `pkexec` can find it (e.g., `/usr/local/bin` if you used `sudo make PREFIX=/usr/local install`, or ensure `~/.local/bin` is in the secure PATH for `pkexec` if installed locally).
    - You might need to configure Polkit rules for `awcc` to allow passwordless execution if desired, though this is an advanced topic. By default, it should ask for your password once per session or per command.

void device_open(void) {
	if (context)
		log_fatal("libusb already initialized");
	if (handle)
		log_fatal("device already opened");

	libusb_init(&context);
	libusb_set_option(context, LIBUSB_OPTION_LOG_LEVEL,
					  LIBUSB_LOG_LEVEL_WARNING);

	libusb_device **devices;
	ssize_t count = libusb_get_device_list(context, &devices);
	if (count < 0)
		log_fatal("get device list");

	libusb_device *device = NULL;
	for (ssize_t i = 0; i < count; i++) {
		struct libusb_device_descriptor descriptor = {0};
		if (libusb_get_device_descriptor(devices[i], &descriptor) != 0)
			log_fatal("get device descriptor");
		if (descriptor.idVendor == 0x187c && (descriptor.idProduct == 0x0551 ||
											  descriptor.idProduct == 0x0550)) {
			device = devices[i];
			break;
		}
	}
	libusb_free_device_list(devices, 1);
	if (device == NULL)
		log_fatal("find device");

	if (libusb_open(device, &handle) != 0)
		log_fatal("open device");
}

void device_acquire(void) {
	if (!handle)
		log_fatal("device not opened");
	if (acquired)
		return;
	if (libusb_kernel_driver_active(handle, 0))
		if (libusb_detach_kernel_driver(handle, 0) != 0)
			log_fatal("detach kernel driver");
	if (libusb_claim_interface(handle, 0) != 0)
		log_fatal("claim interface");
	acquired = true;
}

void device_release(void) {
	if (!handle)
		log_fatal("device not opened");
	if (!acquired)
		return;
	libusb_release_interface(handle, 0);
	libusb_attach_kernel_driver(handle, 0);
	acquired = false;
}

void device_close(void) {
	if (!handle)
		log_fatal("device not opened");
	if (acquired)
		device_release();
	libusb_close(handle);
	handle = NULL;
	libusb_exit(context);
	context = NULL;
}

	if (!acquired)

	unsigned char buffer[33];

	memcpy(buffer, data, length);
	if (libusb_control_transfer(handle, 0x21, 9, 0x202, 0, buffer, 33, 0) != 33)

static const char *get_cpu_model_str() {
	FILE *cpuinfo_fp = fopen("/proc/cpuinfo", "r");

// Gets fan speed for a given type ("cpu" or "gpu") from /sys/class/hwmon
// Returns RPM or -1 if not found/error.
static int get_specific_fan_speed(const char *type_hint) {
	char fan_label_buffer[128];
	char fan_input_path[256];
	char hwmon_name_path[256];
	char hwmon_name_buffer[64];
	int speed = -1;
	int candidate_speed = -1; // For fan1_input if no specific label match

	for (int i = 0; i < 10; ++i) { // Check up to 10 hwmon devices (hwmon0 to hwmon9)
		int current_hwmon_candidate_speed = -1; // Candidate for this specific hwmon iteration

		// For GPU, check if the hwmon device name suggests it's a GPU controller
		if (strcmp(type_hint, "gpu") == 0) {
			snprintf(hwmon_name_path, sizeof(hwmon_name_path),
					 "/sys/class/hwmon/hwmon%d/name", i);
			if (read_line_from_file(hwmon_name_path, hwmon_name_buffer,
									sizeof(hwmon_name_buffer))) {
				for (char *p = hwmon_name_buffer; *p; ++p) {
					*p = tolower(*p); // Convert name to lowercase
				}
				// If name doesn't suggest GPU, skip this hwmon for GPU fan search
				if (!(strstr(hwmon_name_buffer, "gpu") ||
					  strstr(hwmon_name_buffer, "amdgpu") ||
					  strstr(hwmon_name_buffer, "nvidia") ||
					  strstr(hwmon_name_buffer, "radeon"))) {
					continue;
				}
			} else {
				// If name file not readable and we are looking for GPU fan, skip this hwmon
				continue;
			}
		}

		for (int j = 1; j < 5; ++j) { // Check up to 4 fans (fan1 to fan4)
			snprintf(fan_input_path, sizeof(fan_input_path),
					 "/sys/class/hwmon/hwmon%d/fan%d_input", i, j);
			FILE *fp_fan_input = fopen(fan_input_path, "r");
			if (fp_fan_input) {
				int current_read_speed = -1;
				bool read_success =
					(fscanf(fp_fan_input, "%d", &current_read_speed) == 1);
				fclose(fp_fan_input);

				if (read_success) {
					char fan_label_path[256];
					snprintf(fan_label_path, sizeof(fan_label_path),
							 "/sys/class/hwmon/hwmon%d/fan%d_label", i, j);
					if (read_line_from_file(fan_label_path, fan_label_buffer,
											sizeof(fan_label_buffer))) {
						for (char *p = fan_label_buffer; *p; ++p) {
							*p = tolower(*p); // Convert label to lowercase
						}
						if (strstr(fan_label_buffer, type_hint)) {
							return current_read_speed; // Found specific fan, return immediately
						}
					}
					// If it's fan1 and we haven't found a specific label match yet on this hwmon,
					// store as a candidate for this hwmon.
					if (j == 1 && current_hwmon_candidate_speed == -1) {
						current_hwmon_candidate_speed = current_read_speed;
					}
				}
			}
		}
		// If, after checking all fans for this hwmon, we have a candidate (fan1)
		// and haven't found a better global candidate yet, update the global candidate.
		// This prioritizes fan1 from earlier hwmon devices if no label matches.
		if (current_hwmon_candidate_speed != -1 && candidate_speed == -1) {
		    candidate_speed = current_hwmon_candidate_speed;
		}
	}

	if (candidate_speed != -1) {
		return candidate_speed; // Return best candidate (likely a fan1_input) if no specific label found
	}

	// Fallback: try very common paths if the loop didn't yield a result
	if (strcmp(type_hint, "cpu") == 0) {
		const char *common_cpu_paths[] = {
			"/sys/class/hwmon/hwmon0/fan1_input",
			"/sys/class/hwmon/hwmon1/fan1_input",
			"/sys/class/hwmon/hwmon0/fan2_input", // Some systems use fan2 for CPU
			"/sys/class/hwmon/hwmon1/fan2_input",
		};
		for (size_t k = 0; k < sizeof(common_cpu_paths) / sizeof(common_cpu_paths[0]); ++k) {
			FILE *fp_common = fopen(common_cpu_paths[k], "r");
			if (fp_common) {
				if (fscanf(fp_common, "%d", &speed) == 1) {
					fclose(fp_common);
					return speed;
				}
				fclose(fp_common);
			}
		}
	}
	// No highly reliable common paths for GPU fans to add here, as it's very vendor/driver specific.

	return -1; // Not found or error
}

void detectCpuVendor() {
